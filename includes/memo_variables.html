Petit mémo surles variables

Comment déclarer une variable

"use strict";

let maVariable = 1;
console.log(maVariable); // 1
maVariable = 2;
console.log(maVariable); // 2

const maConstante = 1;
console.log(maConstante); // 1
maConstante = 2; // TypeError: Assignment to constant variable.

Traditionnellement, le bon javascripteux n'utilise let que si le contenu d'une variable doit changer. Dans la mesure du possible, le bon javascripteux préfère toujours utiliser const.

Le mot-clef var existe certes toujours, mais uniquement pour des raisons de rétro-compatibilité. Le bon javascripteux ne l'utilise jamais.

Portée des variables

"use strict";

let uneVariable = 1;

function incrementVariable() {
  uneVariable++;
  const uneAutreVariable = 'salut';
}

console.log(uneVariable); // 1
incrementVariable();
console.log(uneVariable); // 2

console.log(uneAutreVariable); // ReferenceError: uneAutreVariable is not defined

Ce qu'il y a entre deux accolades s'appelle un scope. Un scope enfant peut accéder aux variables d'un scope parent, mais pas le contraire.

Les types primitifs

"use strict";

const b = true;
console.log(typeof b);  // boolean

const n = 10;
console.log(typeof n); // number
const f = 10.5;
console.log(typeof f); // number

const s = 'salut';
console.log(typeof s); // string

const u = undefined;
console.log(typeof u); // undefined

Un type primitif est toujours passé par copie à une fonction.

"use strict";

function maFonction(unArgument) {
  unArgument += 10;
  console.log(unArgument)
}
const maVariable = 10;

maFonction(maVariable); // 20
console.log(maVariable); // 10

Tableaux et tableaux indexés (hash) :les objets

"use strict";

const a = [1, 2];
console.log(typeof a); // object
console.log(Array.isArray(a)); // true
console.log(a[0]); // 1

const ai = { a: 1, b: 2};
console.log(typeof ai); // object
console.log(Array.isArray(ai)); // false
console.log(ai.a); // 1
console.log(ai[a]); // 1

const n = null;
console.log(typeof n); // object

Un objet est toujours passé par référence à une fonction.

"use strict";

function maFonction(unArgument) {
  unArgument[0] += 10;
  console.log(unArgument)
}
const maVariable = [10];

maFonction(maVariable); // [ 20 ]
console.log(maVariable); // [ 20 ]
